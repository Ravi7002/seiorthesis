# --- IMPORT SECTION ---
# These are built-in tools (libraries) I need for my script.
import os      # Used for handling file paths and file system tasks.
import json    # Used to convert data (like dictionaries) into text format.
import heapq   # A specialized tool for keeping a list "sorted" so the smallest item is always easy to find (a "heap").
from collections import Counter  # A quick way to count how many times each character appears in a string.

# --- DATA STRUCTURES ---
# This class represents a single "node" in the Huffman Tree.
# Think of it like a branch or a leaf on a tree.
class HeapNode:
    def __init__(self, char, freq):
        self.char = char   # The actual character (like 'A' or 'b').
        self.freq = freq   # How many times this character appeared.
        self.left = None   # Connection to the left child node.
        self.right = None  # Connection to the right child node.

# --- MAIN ENCODER CLASS ---
class HuffmanEncoder:
    def __init__(self):
        # Initializing the workspace.
        self.heap = []             # A priority queue to help build the tree.
        self.codes = {}            # A dictionary (map) to store 'char': '0101' mappings.
        self.reverse_mapping = {}  # A dictionary to store '0101': 'char' mappings.

    # --- Step 1: Count Character Frequencies ---
    def make_frequency_dict(self, text):
        # Uses the 'Counter' tool to create a dictionary of char: frequency.
        # Example: "apple" -> {'a': 1, 'p': 2, 'l': 1, 'e': 1}
        return dict(Counter(text))

    # --- Step 2: Build the Huffman Tree ---
    def make_heap_and_tree(self, frequency):
        self.heap = []
        tie_breaker = 0 # Helps the heap decide if two nodes have the exact same frequency.
        
        # 1. Create a "leaf" for every character and add it to the sorting pool (heap).
        for key in sorted(frequency.keys()):
            node = HeapNode(key, frequency[key])
            # I push a tuple: (frequency, unique_id, node_object).
            # The heap sorts by frequency first.
            heapq.heappush(self.heap, (frequency[key], tie_breaker, node))
            tie_breaker += 1
            
        # 2. Merge the two smallest nodes until only one "root" node remains.
        while len(self.heap) > 1:
            # Pop out the two items with the lowest frequency.
            item1 = heapq.heappop(self.heap)
            item2 = heapq.heappop(self.heap)
            
            node1 = item1[2]
            node2 = item2[2]

            # Create a new "parent" node that combines their frequencies.
            merged = HeapNode(None, node1.freq + node2.freq)
            merged.left = node1
            merged.right = node2
            
            # Put this new branch back into the heap.
            heapq.heappush(self.heap, (merged.freq, tie_breaker, merged))
            tie_breaker += 1

    # --- Step 3: Assign Binary Codes (0s and 1s) ---
    def make_codes_helper(self, root, current_code):
        # Recursive function: walks down the tree.
        if root is None:
            return

        # If I reach a leaf (a node with a character), save its path.
        if root.char is not None:
            self.codes[root.char] = current_code
            self.reverse_mapping[current_code] = root.char
            return

        # Go left (add a '0') and then go right (add a '1').
        self.make_codes_helper(root.left, current_code + "0")
        self.make_codes_helper(root.right, current_code + "1")

    def make_codes(self):
        if not self.heap:
            return
            
        # The last item in the heap is the "root" of the completed tree.
        root_item = heapq.heappop(self.heap)
        root = root_item[2]
        
        current_code = ""
        
        # Special case: If the file only has one type of character.
        if root.char is not None: 
             self.codes[root.char] = "0"
             self.reverse_mapping["0"] = root.char
             return

        # Start the recursive journey to map every character.
        self.make_codes_helper(root, current_code)
    
    def run_tree_building(self, frequency):
         # Helper to reset the state and run the tree building logic.
         self.heap = []
         self.codes = {}
         self.reverse_mapping = {}
         
         self.make_heap_and_tree(frequency)
         self.make_codes()

    # --- Step 4: Convert Text to Bitstring ---
    def get_encoded_text(self, text):
        # Replace every letter with its new short binary code.
        encoded_text = ""
        for char in text:
            encoded_text += self.codes[char]
        return encoded_text

    # --- Step 5: Padding (Making it fit into 8-bit bytes) ---
    def pad_encoded_text(self, encoded_text):
        # Files are saved in bytes (8 bits). If the code is 11 bits, I need to add 5 filler bits.
        extra_padding = 8 - len(encoded_text) % 8
        for i in range(extra_padding):
            encoded_text += "0"

        # I also store how many filler bits I added so I can remove them later.
        # This info is saved as the first 8 bits of the actual data.
        padded_info = "{0:08b}".format(extra_padding)
        encoded_text = padded_info + encoded_text
        return encoded_text

    # --- Step 6: Convert String of "0" and "1" into Real Binary ---
    def get_byte_array(self, padded_encoded_text):
        if len(padded_encoded_text) % 8 != 0:
            print("Padding error")
            exit(0)

        b = bytearray()
        # Grab 8 "bits" at a time and convert them to a real number (0-255).
        for i in range(0, len(padded_encoded_text), 8):
            byte = padded_encoded_text[i:i+8]
            b.append(int(byte, 2)) # base 2 conversion
        return b

    # --- MAIN COMPRESSION EXECUTION ---
    def compress(self, input_path, output_path=None):
        # Set default output name if none provided.
        if output_path is None:
            filename, file_extension = os.path.splitext(input_path)
            output_path = filename + ".bin"

        # Read the source file.
        with open(input_path, 'r', encoding='utf-8') as file:
            text = file.read()

        # Don't try to compress air!
        if not text:
            print(f"Skipping empty file: {input_path}")
            return None

        try:
            # 1. Analyze the text.
            frequency = self.make_frequency_dict(text)
            self.run_tree_building(frequency)
    
            # 2. Translate text to codes.
            encoded_text = self.get_encoded_text(text)
            padded_encoded_text = self.pad_encoded_text(encoded_text)
            binary_data = self.get_byte_array(padded_encoded_text)
            
            # 3. Save the results.
            # I save a "header" (the frequency map) so the decoder knows how to rebuild the tree.
            header = json.dumps(frequency)
            header_bytes = bytearray(header + "\n", 'utf-8')
    
            with open(output_path, 'wb') as output:
                output.write(header_bytes) # Write the mapping first.
                output.write(binary_data)   # Write the compressed data second.
            
            print(f"Compressed {input_path} -> {output_path}")
            return output_path
        except Exception as e:
            print(f"Error compressing {input_path}: {e}")
            return None

# THIS BLOCK RUNS IF YOU EXECUTE THE SCRIPT DIRECTLY
if __name__ == "__main__":
    import sys
    # Check if a filename was provided in the command line.
    if len(sys.argv) > 1:
        path = sys.argv[1]
        encoder = HuffmanEncoder()
        encoder.compress(path)
    else:
        print("Usage: python Encoder.py <path_to_file>")
